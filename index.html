// 統合版のアクション処理ロジック async function resolveBattle() { if(myHands.length === 0 || s.isPlaying) return; s.isPlaying = true; for (let i = 0; i < myHands.length; i++) { let cmd = myHands[i]; // 仕様書 ID 03: 地上パンチの2段切り替え if (cmd === 'punch') { s.pCur = (i % 2 === 0) ? 'punch' : 'punch2'; s.shake = 20; // 仕様書 ID 07: 振動 s.enHP -= 10; } // 仕様書 ID 08: アッパーの残像と空中挙動 else if (cmd === 'upper') { s.isAir = true; s.vy = -20; s.pCur = 'upper'; spawnAfterImage(s.x, s.y); // 残像生成 } else { s.pCur = 'guard'; } document.getElementById('enBar').style.width = s.enHP + "%"; await new Promise(r => setTimeout(r, 600)); s.pCur = 'player'; } s.isPlaying = false; resetCmd(); } // 残像生成関数 function spawnAfterImage(x, y) { s.afterImages.push({x, y, alpha: 0.5}); } // 描画ループ内での演出反映 function loop() { ctx.fillStyle = '#0a192f'; ctx.fillRect(0, 0, canvas.width, canvas.height); // 残像の描画とフェードアウト s.afterImages.forEach((img, i) => { ctx.globalAlpha = img.alpha; ctx.drawImage(imgs.dash, img.x, img.y, 240, 240); img.alpha -= 0.1; if(img.alpha <= 0) s.afterImages.splice(i, 1); }); ctx.globalAlpha = 1.0; // 振動（仕様書 07） let off = (Math.random()-0.5) * s.shake; ctx.drawImage(imgs[s.pCur], s.x + off, s.y, 240, 240); s.shake *= 0.8; // ...以降の敵描画処理 }
