async function resolveBattle() { if(myHands.length === 0 || s.isPlaying) return; s.isPlaying = true; // 座標初期化：中央での攻防に備える s.y = 200; e.y = 200; // 1. 開始：中央へダッシュ await animateVersus(300, 460, 300, 'dash', 'dash'); let airCombo = false, airStep = 0; for (let i = 0; i < myHands.length; i++) { let p = myHands[i]; // --- 空中コンボ処理 --- if (airCombo) { airStep++; if (p === 'punch' && airStep <= 3) { // 3回目（メテオ）の判定 if (airStep === 3) { s.pCur = 'knock'; e.pCur = 'damage'; await sleep(500); s.pCur = 'knock2'; applyDmg(e, DMG.METEOR); // メテオダメージ await sleep(600); airCombo = false; } else { s.pCur = (airStep === 1 ? 'punch' : 'punch2'); applyDmg(e, DMG.UPPER); await sleep(300); } } else { // 空中でパンチ以外が来た場合、落下してコンボ終了 airCombo = false; s.y = 200; e.y = 200; } continue; } // --- 地上攻防（中央） --- if (p === 'upper') { s.pCur = 'upper'; e.pCur = 'damage'; applyDmg(e, DMG.UPPER); await sleep(300); e.y = 50; // 敵は必ず浮く if (myHands[i+1] === 'punch') { s.pCur = 'upper2'; s.y = 50; airCombo = true; airStep = 0; await sleep(500); } else { // コンボへ行かない場合、敵は自然落下 await sleep(300); e.y = 200; e.pCur = 'knock2'; await sleep(500); } } else if (p === 'guard') { s.pCur = 'guard'; s.shake = 20; await sleep(400); } else { s.pCur = 'dash'; applyDmg(e, DMG.PUNCH); await sleep(300); } } // 2. 終了：元位置へ帰還 await animateVersus(100, 660, 300, 'dash', 'dash'); s.pCur = 'player'; e.pCur = 'player'; s.y = 200; e.y = 200; s.isPlaying = false; resetCmd(); }
