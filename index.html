(Math.random()-0.5)*20 : 0) : o.x; if(i === 1) { // 敵は反転させず、位置を固定して描画 ctx.drawImage(imgs[o.pCur], 660, o.y + (o.yOff||0), 200, 200); } else { ctx.drawImage(imgs[o.pCur], x, o.y + (o.yOff||0), 200, 200); } } }); requestAnimationFrame(loop); }
